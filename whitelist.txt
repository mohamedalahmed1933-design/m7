local ESP_ENABLED = true          
local MAX_DISTANCE = 3000         
local UPDATE_INTERVAL = 0.02 

local RARITY_COLORS = {
    ["Common"] = Color3.fromRGB(255, 255, 255),    
    ["Uncommon"] = Color3.fromRGB(99, 255, 52),   
    ["Rare"] = Color3.fromRGB(51, 170, 255),       
    ["Epic"] = Color3.fromRGB(237, 44, 255),       
    ["Legendary"] = Color3.fromRGB(255, 150, 0),   
    ["Omega"] = Color3.fromRGB(255, 20, 51),

local Loot_Cache = {}

-- 🛠️ دالة مخصصة لفك شفرة أسلحة ولوت Bloxspin وقراءة الأسماء الحقيقية بدلاً من الأرقام
local function getCleanName(tool)
    if not tool then return nil end
    
    -- 1. محاولة القراءة من الـ Attributes الشائعة في نظام اللعبة للأسلحة واللوت الأرضي
    local attrName = tool:GetAttribute("DisplayName") or tool:GetAttribute("WeaponName") or tool:GetAttribute("ItemName") or tool:GetAttribute("Name")
    if attrName and tostring(attrName) ~= "" and not tonumber(attrName) then
        return tostring(attrName)
    end

    -- 2. محاولة البحث عن كائن نصي خفي (StringValue) يحتوي على الاسم الحقيقي داخل السلاح الساقط
    for _, child in ipairs(tool:GetChildren()) do
        if child:IsA("StringValue") and child.Value ~= "" and not tonumber(child.Value) then
            local cName = child.Name:lower()
            if cName:find("name") or cName:find("weapon") or cName:find("display") or cName:find("tool") or cName:find("item") then
                return child.Value
            end
        end
    end

    -- 3. تنظيف أولي لاسم الأداة الأصلي من أي أرقام مدمجة أو زوائد برمجية
    local originalName = tool.Name
    local cleaned = originalName:gsub("%d+", ""):gsub("_", " "):gsub("-", " "):gsub("[%s]+", " "):match("^%s*(.-)%s*\$")
    
    -- إذا كان الاسم يحتوي على أرقام فقط (كود برمي) ولم يجد اسماً حقيقياً بالداخل
    if tonumber(originalName) and (not cleaned or cleaned == "") then
        return "Weapon ID: " .. originalName 
    end

    if not cleaned or cleaned == "" then cleaned = originalName end
    return (cleaned:gsub("(%a)([%w]*)", function(first, rest) return first:upper()..rest:lower() end))
end

-- === 👤 1️⃣ تحديث كاشف اللاعبين (الأسماء باللون الأبيض) ===
local function updateESP(player)
    local data = ESP_Cache[player]
    if not data then return end

    local char = player.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    local head = char and char:FindFirstChild("Head")
    local humanoid = char and char:FindFirstChildOfClass("Humanoid")

    if not char or not root or not head or not humanoid or humanoid.Health <= 0 then
        if data.Billboard then data.Billboard.Enabled = false end
        if data.Box then data.Box.Adornee = nil end
        return
    end

    local lChar = LocalPlayer.Character
    local lRoot = lChar and lChar:FindFirstChild("HumanoidRootPart")
    if not lRoot then 
        if data.Billboard then data.Billboard.Enabled = false end
        if data.Box then data.Box.Adornee = nil end
        return 
    end

    local dist = math.floor((lRoot.Position - root.Position).Magnitude)
    if dist > MAX_DISTANCE then
        if data.Billboard then data.Billboard.Enabled = false end
        if data.Box then data.Box.Adornee = nil end
        return
    end

    if data.Box then data.Box.Adornee = char end

    local container = data.Billboard and data.Billboard:FindFirstChild("EspContainer")
    if container then
        container:ClearAllChildren()

        local layout = Instance.new("UIListLayout", container)
        layout.HorizontalAlignment = Enum.HorizontalAlignment.Center
        layout.VerticalAlignment = Enum.VerticalAlignment.Bottom
        layout.Padding = UDim.new(0, 2)

        local infoLbl = Instance.new("TextLabel", container)
        infoLbl.Size = UDim2.new(0, 200, 0, 24)
        infoLbl.BackgroundTransparency = 1
        infoLbl.Text = string.format("%s\n[%d m]  HP: %d/%d", player.Name, dist, math.floor(humanoid.Health), math.floor(humanoid.MaxHealth))
        infoLbl.Font = Enum.Font.SourceSansBold
        infoLbl.TextSize = 14
        infoLbl.TextColor3 = Color3.fromRGB(255, 255, 255) -- اسم اللاعب باللون الأبيض
        infoLbl.TextStrokeTransparency = 0.1
        infoLbl.TextStrokeColor3 = Color3.new(0, 0, 0)

        if humanoid.Health < (humanoid.MaxHealth * 0.25) then
            infoLbl.TextColor3 = Color3.fromRGB(255, 50, 50) 
        end

        local tools = {}
        if player:FindFirstChild("Backpack") then
            for _, t in ipairs(player.Backpack:GetChildren()) do
                if t:IsA("Tool") and t.Name:lower() ~= "fists" then table.insert(tools, t) end
            end
        end
        for _, t in ipairs(char:GetChildren()) do
            if t:IsA("Tool") and t.Name:lower() ~= "fists" then table.insert(tools, t) end
        end

        for _, tool in ipairs(tools) do
            local name = getCleanName(tool)
            if name then
                local lbl = Instance.new("TextLabel", container)
                lbl.Size = UDim2.new(0, 200, 0, 14)
                lbl.BackgroundTransparency = 1
                lbl.Text = "[ " .. name .. " ]"
                lbl.Font = Enum.Font.SourceSansBold
                lbl.TextSize = 12
                lbl.TextStrokeTransparency = 0.2
                lbl.TextStrokeColor3 = Color3.new(0,0,0)

                local rarity = tool:GetAttribute("RarityName") or tool:GetAttribute("Rarity")
                lbl.TextColor3 = RARITY_COLORS[rarity] or Color3.fromRGB(0, 255, 255)
            end
        end
    end

    if data.Billboard then
        data.Billboard.Enabled = ESP_ENABLED
        data.Billboard.Adornee = head
    end
end

-- === 📦 2️⃣ نظام كشف اللوت والصناديق تلقائياً (بدون أرقام) ===
local function checkAndTrackLoot(object)
    if not object:IsA("Model") and not object:IsA("Part") and not object:IsA("MeshPart") and not object:IsA("Tool") then return end
    
    local nameLower = object.Name:lower()
    local isChest = nameLower:find("chest") or nameLower:find("treasure") or nameLower:find("box") or nameLower:find("safe")
    local isDroppedItem = object:IsA("Tool") or object:FindFirstChildOfClass("TouchTransmitter") or nameLower:find("item") or nameLower:find("loot") or nameLower:find("ground") or object:FindFirstChild("Handle")

    if (isChest or isDroppedItem) and not Loot_Cache[object] then
        local primaryPart = object:IsA("Model") and (object.PrimaryPart or object:FindFirstChildWhichIsA("BasePart")) or object
        if not primaryPart then return end

        -- استخدام الدالة الذكية لفك شفرة الاسم الحقيقي للوت الأرضي
        local cleanName = isChest and "Treasure Chest 🎁" or getCleanName(object)
        local displayColor = isChest and Color3.fromRGB(255, 215, 0) or Color3.fromRGB(0, 255, 100) -- ذهبي للصناديق وأخضر للوت

        local bb = Instance.new("BillboardGui")
        bb.Name = "LootESP_" .. object.Name
        bb.Size = UDim2.new(0, 150, 0, 40)
        bb.AlwaysOnTop = true
        bb.ExtentsOffset = Vector3.new(0, 2, 0)
        
        local successUi, _ = pcall(function() bb.Parent = CoreGui end)
        if not successUi then bb.Parent = LocalPlayer:WaitForChild("PlayerGui") end

        local lbl = Instance.new("TextLabel", bb)
        lbl.Size = UDim2.new(1, 0, 1, 0)
        lbl.BackgroundTransparency = 1
        lbl.Font = Enum.Font.SourceSansBold
        lbl.TextSize = 13
        lbl.TextColor3 = displayColor
        lbl.TextStrokeTransparency = 0.2
        lbl.TextStrokeColor3 = Color3.new(0, 0, 0)

        local selectionBox = Instance.new("SelectionBox")
        selectionBox.Color3 = displayColor
        selectionBox.LineThickness = 0.04
        selectionBox.Adornee = object
        selectionBox.Parent = workspace

        Loot_Cache[object] = { Billboard = bb, Label = lbl, Box = selectionBox, Part = primaryPart, Name = cleanName }
    end
end

local function updateLootESP()
    local lChar = LocalPlayer.Character
    local lRoot = lChar and lChar:FindFirstChild("HumanoidRootPart")

    for obj, data in pairs(Loot_Cache) do
        if not obj or not obj.Parent or not data.Part or not data.Part.Parent then
            if data.Billboard then data.Billboard:Destroy() end
            if data.Box then data.Box:Destroy() end
            Loot_Cache[obj] = nil
        else
            if lRoot and ESP_ENABLED then
                local dist = math.floor((lRoot.Position - data.Part.Position).Magnitude)
                if dist <= MAX_DISTANCE then
                    data.Billboard.Enabled = true
                    data.Box.Visible = true
                    data.Label.Text = string.format("%s\n[%d m]", data.Name, dist)
                else
                    data.Billboard.Enabled = false
                    data.Box.Visible = false
                end
            else
                data.Billboard.Enabled = false
                data.Box.Visible = false
            end
        end
    end
end

-- مسح الماب وتتبع اللوت تلقائياً
for _, obj in ipairs(workspace:GetDescendants()) do pcall(checkAndTrackLoot, obj) end
workspace.DescendantAdded:Connect(function(obj) pcall(checkAndTrackLoot, obj) end)
