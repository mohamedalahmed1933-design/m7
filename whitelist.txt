local Players = game:GetService("Players")
local HttpService = game:GetService("HttpService")
local RunService = game:GetService("RunService")
local CoreGui = game:GetService("CoreGui")
local LocalPlayer = Players.LocalPlayer

-- ==========================================
-- === 🌟 إعدادات الكاشف المتطور (ESP) ===
-- ==========================================
local ESP_ENABLED = true          
local MAX_DISTANCE = 3000         
local UPDATE_INTERVAL = 0.02 

local RARITY_COLORS = {
    ["Common"] = Color3.fromRGB(255, 255, 255),    
    ["Uncommon"] = Color3.fromRGB(99, 255, 52),   
    ["Rare"] = Color3.fromRGB(51, 170, 255),       
    ["Epic"] = Color3.fromRGB(237, 44, 255),       
    ["Legendary"] = Color3.fromRGB(255, 150, 0),   
    ["Omega"] = Color3.fromRGB(255, 20, 51),       
}

local ESP_Cache = {}

-- دالة متطورة لتنظيف أسماء الأسلحة من الأرقام والرموز
local function getCleanName(tool)
    if not tool or not tool.Name then return nil end
    local originalName = tool.Name
    
    -- تنظيف الاسم من الأرقام والرموز تماماً
    local cleaned = originalName:gsub("%d+", ""):gsub("_", " "):gsub("-", " "):gsub("[%s]+", " "):match("^%s*(.-)%s*$")
    
    -- إذا أصبح الاسم فارغاً بسبب الحذف، نرجع الاسم الأصلي لحمايته
    if not cleaned or cleaned == "" then 
        cleaned = originalName 
    end

    -- التحقق من أدوات الصيد كدعم إضافي
    local lowerClean = cleaned:lower()
    if lowerClean:find("fishing") or lowerClean:find("rod") or lowerClean:find("canne") then
        if lowerClean:find("ultimate") or lowerClean:find("ult") then return "Ultimate Fishing Rod"
        elseif lowerClean:find("advanced") then return "Advanced Fishing Rod"
        elseif lowerClean:find("pro") then return "Pro Fishing Rod"
        else return "Regular Fishing Rod" end
    end
    
    -- تكبير أول حرف من كل كلمة ليصبح المظهر احترافي
    return (cleaned:gsub("(%a)([%w]*)", function(first, rest) return first:upper()..rest:lower() end))
end

local function updateESP(player)
    local data = ESP_Cache[player]
    if not data then return end

    local char = player.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    local head = char and char:FindFirstChild("Head")
    local humanoid = char and char:FindFirstChildOfClass("Humanoid")

    if not char or not root or not head or not humanoid or humanoid.Health <= 0 then
        if data.Billboard then data.Billboard.Enabled = false end
        if data.Box then data.Box.Adornee = nil end
        return
    end

    local lChar = LocalPlayer.Character
    local lRoot = lChar and lChar:FindFirstChild("HumanoidRootPart")
    if not lRoot then 
        if data.Billboard then data.Billboard.Enabled = false end
        if data.Box then data.Box.Adornee = nil end
        return 
    end

    local dist = math.floor((lRoot.Position - root.Position).Magnitude)
    if dist > MAX_DISTANCE then
        if data.Billboard then data.Billboard.Enabled = false end
        if data.Box then data.Box.Adornee = nil end
        return
    end

    -- === 1️⃣ تحديث المربع المحيط باللاعب ===
    if data.Box then
        data.Box.Adornee = char
    end

    -- === 2️⃣ تحديث النصوص (الاسم باللون الأبيض، المسافة، الدم) ===
    local container = data.Billboard and data.Billboard:FindFirstChild("EspContainer")
    if container then
        container:ClearAllChildren()

        local layout = Instance.new("UIListLayout", container)
        layout.HorizontalAlignment = Enum.HorizontalAlignment.Center
        layout.VerticalAlignment = Enum.VerticalAlignment.Bottom
        layout.Padding = UDim.new(0, 2)

        local infoLbl = Instance.new("TextLabel", container)
        infoLbl.Size = UDim2.new(0, 200, 0, 24)
        infoLbl.BackgroundTransparency = 1
        infoLbl.Text = string.format("%s\n[%d m]  HP: %d/%d", player.Name, dist, math.floor(humanoid.Health), math.floor(humanoid.MaxHealth))
        infoLbl.Font = Enum.Font.SourceSansBold
        infoLbl.TextSize = 14
        infoLbl.TextColor3 = Color3.fromRGB(255, 255, 255) -- ⚪ اسم اللاعب باللون الأبيض النقي
        infoLbl.TextStrokeTransparency = 0.1
        infoLbl.TextStrokeColor3 = Color3.new(0, 0, 0)

        -- تغيير اللون فقط إذا كان الدم منخفضاً جداً كتحذير
        if humanoid.Health < (humanoid.MaxHealth * 0.25) then
            infoLbl.TextColor3 = Color3.fromRGB(255, 50, 50) -- أحمر عند الاحتضار
        end

        -- === 3️⃣ تحديث أسلحة وأدوات اللاعب (بدون أرقام) ===
        local tools = {}
        if player:FindFirstChild("Backpack") then
            for _, t in ipairs(player.Backpack:GetChildren()) do
                if t:IsA("Tool") and t.Name:lower() ~= "fists" then table.insert(tools, t) end
            end
        end
        for _, t in ipairs(char:GetChildren()) do
            if t:IsA("Tool") and t.Name:lower() ~= "fists" then table.insert(tools, t) end
        end

        for _, tool in ipairs(tools) do
            local name = getCleanName(tool)
            if name then
                local lbl = Instance.new("TextLabel", container)
                lbl.Size = UDim2.new(0, 200, 0, 14)
                lbl.BackgroundTransparency = 1
                lbl.Text = "[ " .. name .. " ]"
                lbl.Font = Enum.Font.SourceSansBold
                lbl.TextSize = 12
                lbl.TextStrokeTransparency = 0.2
                lbl.TextStrokeColor3 = Color3.new(0,0,0)

                local rarity = tool:GetAttribute("RarityName") or tool:GetAttribute("Rarity")
                lbl.TextColor3 = RARITY_COLORS[rarity] or Color3.fromRGB(0, 255, 255)
            end
        end
    end

    if data.Billboard then
        data.Billboard.Enabled = ESP_ENABLED
        data.Billboard.Adornee = head
    end
end

local function createESP(player)
    if player == LocalPlayer then return end

    local function setup(char)
        char:WaitForChild("HumanoidRootPart", 5)
        char:WaitForChild("Head", 5)

        if ESP_Cache[player] then 
            if ESP_Cache[player].Billboard then ESP_Cache[player].Billboard:Destroy() end
            if ESP_Cache[player].Box then ESP_Cache[player].Box:Destroy() end
        end

        local bb = Instance.new("BillboardGui")
        bb.Name = "PlayerESP_" .. player.Name
        bb.Size = UDim2.new(0, 200, 0, 150)
        bb.AlwaysOnTop = true
        bb.ExtentsOffset = Vector3.new(0, 3.5, 0)
        bb.Enabled = false
        
        local successUi, _ = pcall(function() bb.Parent = CoreGui end)
        if not successUi then bb.Parent = LocalPlayer:WaitForChild("PlayerGui") end

        local container = Instance.new("Frame", bb)
        container.Name = "EspContainer"
        container.Size = UDim2.new(1, 0, 1, 0)
        container.BackgroundTransparency = 1

        local box = Instance.new("SelectionBox")
        box.Name = "BoxESP_" .. player.Name
        box.Color3 = Color3.fromRGB(255, 0, 0) 
        box.LineThickness = 0.05
        box.Parent = workspace

        ESP_Cache[player] = { Billboard = bb, Box = box }
    end

    if player.Character then task.spawn(setup, player.Character) end
    player.CharacterAdded:Connect(setup)
end

for _, p in ipairs(Players:GetPlayers()) do createESP(p) end
Players.PlayerAdded:Connect(createESP)

Players.PlayerRemoving:Connect(function(p)
    if ESP_Cache[p] then
        if ESP_Cache[p].Billboard then ESP_Cache[p].Billboard:Destroy() end
        if ESP_Cache[p].Box then ESP_Cache[p].Box:Destroy() end
        ESP_Cache[p] = nil
    end
end)

task.spawn(function()
    while true do
        for p, _ in pairs(ESP_Cache) do
            pcall(function() updateESP(p) end)
        end
        task.wait(UPDATE_INTERVAL)
    end
end)

print("Rayiz Clean ESP Fully Loaded! 🚀")
